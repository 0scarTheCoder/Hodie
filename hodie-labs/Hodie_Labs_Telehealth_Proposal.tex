\section{The Problem}

\subsection{Current Australian Healthcare System Challenges}

\subsubsection{For Patients}

\begin{itemize}[leftmargin=*]
    \item \textbf{Long wait times:} Average 2-3 weeks for GP appointments in metro areas
    \item \textbf{Time burden:} 60-120 minutes total (travel + waiting + consult)
    \item \textbf{Fragmented care:} Medical history repeated at every visit
    \item \textbf{Limited access:} GP shortages in regional/rural areas
    \item \textbf{After-hours gaps:} Limited bulk-billed options outside business hours
    \item \textbf{Cost barriers:} Non-bulk-billed consults cost \$80-150 out-of-pocket
\end{itemize}

\subsubsection{For GPs}

\begin{itemize}[leftmargin=*]
    \item \textbf{High overhead:} Rent, staff, utilities consume 60-70\% of revenue
    \item \textbf{Low profit margins:} Only 30-35\% of Medicare rebate retained
    \item \textbf{Administrative burden:} 30-40\% of time on non-clinical tasks
    \item \textbf{Burnout:} Long hours, patient overload, paperwork stress
    \item \textbf{Information gaps:} Incomplete patient history, missing test results
    \item \textbf{Location-bound:} Must be physically present at clinic
\end{itemize}

\subsection{Eligibility Requirements}

\subsubsection{For GPs to Bill Medicare}

\begin{enumerate}[leftmargin=*]
    \item \textbf{Provider Number} - Valid Medicare provider registration
    \item \textbf{AHPRA Registration} - Current medical practitioner registration
    \item \textbf{Medical Indemnity Insurance} - Professional indemnity cover (MIPS/AVANT/MDA National)
    \item \textbf{Patient Relationship} - For some items, prior treating relationship required (waived for initial consults)
    \item \textbf{Clinical Appropriateness} - Telehealth must be clinically appropriate for the condition
    \item \textbf{Documentation} - Adequate clinical notes and record-keeping
\end{enumerate}

\subsubsection{For Patients}

\begin{enumerate}[leftmargin=*]
    \item \textbf{Medicare Card} - Valid Medicare or DVA card
    \item \textbf{Identity Verification} - Driver's license or passport verification
    \item \textbf{Consent} - Informed consent for telehealth consultation
    \item \textbf{Technology Access} - Device with camera/microphone and stable internet
\end{enumerate}

\section{Regulatory Compliance}

\subsection{AHPRA Requirements}

\textbf{GP Verification Process:}
\begin{enumerate}
    \item AHPRA registration number verification (via AHPRA public register API)
    \item Practicing certificate validation
    \item Medical indemnity insurance proof (MIPS/AVANT/MDA National)
    \item Identity verification (100-point check)
    \item Qualifications verification (medical degree, fellowship)
\end{enumerate}

\textbf{Ongoing Compliance:}
\begin{itemize}
    \item Annual registration renewal verification
    \item Continuing professional development (CPD) tracking
    \item Complaints and notification monitoring
    \item Practice audits (random selection)
\end{itemize}

\subsection{RACGP Telehealth Standards}

\textbf{Clinical Guidelines:}
\begin{enumerate}
    \item \textbf{Patient Identification} - Photo ID verification
    \item \textbf{Informed Consent} - Written consent for telehealth
    \item \textbf{Clinical Appropriateness} - Telehealth suitable for condition
    \item \textbf{Emergency Protocols} - Escalation pathways documented
    \item \textbf{Clinical Handover} - Documentation sent to usual GP
    \item \textbf{Prescribing Standards} - Appropriate telehealth prescribing
\end{enumerate}

\textbf{Platform Requirements:}
\begin{itemize}
    \item Secure video platform (encrypted)
    \item Clear audio and video quality standards
    \item Technical support for patients
    \item Fallback to phone consult if video fails
\end{itemize}

\subsection{Privacy Act 1988 \& My Health Records}

\textbf{Data Protection:}
\begin{itemize}
    \item Patient consent for data collection and use
    \item Right to access medical records
    \item Right to correct inaccurate data
    \item Right to delete data (subject to legal requirements)
    \item Data breach notification (within 72 hours)
\end{itemize}

\textbf{My Health Record Integration:}
\begin{itemize}
    \item Optional integration for GPs to upload consultation notes
    \item Shared health summary
    \item Event summary for hospital admissions
    \item Prescription and dispense records
\end{itemize}

\subsection{Medicare Compliance}

\textbf{Bulk-Billing Requirements:}
\begin{enumerate}
    \item GP must accept Medicare rebate as full payment
    \item No additional charges to patient (must be \$0 out-of-pocket)
    \item Informed consent from patient for bulk-billing
    \item Correct item number selection and documentation
\end{enumerate}

\textbf{Known Patient Criteria:}
\begin{itemize}
    \item For some telehealth items, GP must have prior treating relationship
    \item \textbf{Exception:} Initial consultation items (91800, 91801) do not require prior relationship
    \item Hodie Labs facilitates new patient consultations (compliant)
\end{itemize}

\textbf{Documentation Requirements:}
\begin{itemize}
    \item Date, time, and duration of consultation
    \item Patient identification details
    \item Clinical assessment and management plan
    \item Item number and justification
    \item GP signature and provider number
\end{itemize}

\subsection{Prescribing Regulations}

\textbf{Schedule 4 Drugs (Prescription-Only):}
\begin{itemize}
    \item Can be prescribed via telehealth with appropriate clinical assessment
    \item GP must have patient history and examination data
    \item Electronic prescribing via eRx system
\end{itemize}

\textbf{Schedule 8 Drugs (Controlled):}
\begin{itemize}
    \item \textbf{Restricted} for first-time patients via telehealth
    \item Requires in-person assessment or existing treating relationship
    \item Real-time prescription monitoring (RTPM) system
    \item State-specific regulations (vary by jurisdiction)
\end{itemize}

\textbf{Hodie Labs Policy:}
\begin{itemize}
    \item Schedule 8 prescriptions only for existing patients
    \item Automated alerts for high-risk prescriptions
    \item Integration with RTPM systems (SafeScript Victoria, etc.)
\end{itemize}

\subsection{Strategic Recommendation}

\textbf{Proceed with telehealth integration.}

\textbf{Rationale:}
\begin{enumerate}[leftmargin=*]
    \item \textbf{Strong unit economics:} 88\% profit margin for GPs, 85\% for Hodie Labs
    \item \textbf{Regulatory compliant:} Platform fee model avoids fee-splitting prohibition
    \item \textbf{Market gap:} No competitor integrates AI health data with telehealth
    \item \textbf{Synergistic:} Increases health data platform stickiness and retention
    \item \textbf{Scalable:} Marginal cost near zero, network effects
    \item \textbf{Fast payback:} 13.8-month payback period, 89\% IRR
